import { useState } from 'react';
import { Props } from 'docz';

# LoadingSpin

## Propriedades

<Props of={LoadingSpin} />

## Componente

<Playground>
	{() => {
		const [loading, setLoading] = useState(false);
    const onClick = () => {
      setLoading(true);
      setTimeout(() => setLoading(false), 2000);
    };
		return <>
      <button onClick={onClick}>Mostrar Loading Spin</button>
      <LoadingSpin loading={loading} />
    </>;
	}}
</Playground>
